L1 = np.array([30, 65, 70, 50, 30, -85, 100])

# count, bins, patches = pyplot.hist(L1, bins=40, density=True, alpha=0.6, color='b', edgecolor='k') # unscaled histogram
count, bins = np.histogram(L1, bins=42, density=True)

kde = gaussian_kde(L1, bw_method=0.10) # define kernel density function
x = np.linspace(-100, 100, 100) # define range
kdeValues = kde(x)

binWidth = np.diff(bins)
scaledCounts = count / binWidth # scale histogram bars

pyplot.bar(bins[:-1], scaledCounts, width=binWidth, color='b', edgecolor='k', alpha=0.6, align='edge')
pyplot.plot(x, kdeValues, color='r', linewidth=2, label='response distribution')
pyplot.title("Likert Scale Response Probability Distribution for Question #2")
pyplot.xlabel("Likert Scale Value")
pyplot.ylabel("Response Probability Density")
pyplot.xticks([-100,0,100], ['Strongly Disagree (-100)','Neutral (0)','Strongly Agree (+100)'])
pyplot.axvline(x=L1mean, color='k', linestyle='--', label='mean')
pyplot.axvline(x=L1median, color='m', linestyle='--', label='median')
pyplot.legend()
pyplot.show()
